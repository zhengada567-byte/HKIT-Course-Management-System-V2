      if((gx/40+gy/40)%2===0)ctx.fillRect(gx,gy,40,40);
    // 移动
    if(worldKeys['ArrowUp']||worldKeys['w'])   worldPlayer.y=Math.max(worldPlayer.size,worldPlayer.y-SPD);
    if(worldKeys['ArrowDown']||worldKeys['s']) worldPlayer.y=Math.min(H-worldPlayer.size-70,worldPlayer.y+SPD);
    if(worldKeys['ArrowLeft']||worldKeys['a']) worldPlayer.x=Math.max(worldPlayer.size,worldPlayer.x-SPD);
    if(worldKeys['ArrowRight']||worldKeys['d'])worldPlayer.x=Math.min(W-worldPlayer.size,worldPlayer.x+SPD);
    // 绘制道具
    worldItems.forEach(item=>{
      if(item.collected)return;
      if(item.type==='coin'){
        ctx.beginPath();ctx.arc(item.x,item.y,item.r,0,Math.PI*2);
        ctx.fillStyle='#ffcc00';ctx.fill();
        ctx.fillStyle='#000';ctx.font='8px Arial';ctx.textAlign='center';ctx.fillText(item.val,item.x,item.y+3);
      }else if(item.type==='chest'){
        ctx.fillStyle='#884400';ctx.fillRect(item.x-item.r,item.y-item.r,item.r*2,item.r*2);
        ctx.fillStyle='#fff';ctx.font='11px Arial';ctx.textAlign='center';ctx.fillText('📦',item.x,item.y+4);
      }else if(item.type==='statue'){
        ctx.beginPath();ctx.arc(item.x,item.y,item.r,0,Math.PI*2);
        ctx.fillStyle='#888';ctx.fill();
        ctx.fillStyle='#fff';ctx.font='12px Arial';ctx.textAlign='center';ctx.fillText('🗿',item.x,item.y+5);
      }else if(item.type==='buff'){
        const icons={dmgUp:'⚔️',shield:'🛡️',cdHalf:'⚡'};
        const colors={dmgUp:'#ff4444',shield:'#4444ff',cdHalf:'#ffff00'};
        ctx.beginPath();ctx.arc(item.x,item.y,item.r,0,Math.PI*2);
        ctx.fillStyle=colors[item.buffType]+'55';ctx.fill();
        ctx.strokeStyle=colors[item.buffType];ctx.lineWidth=2;ctx.stroke();
        ctx.font='12px Arial';ctx.textAlign='center';ctx.fillText(icons[item.buffType],item.x,item.y+4);
      }else if(item.type==='hidden'){
        // 隐藏点：微弱闪烁
        const alpha=0.15+Math.sin(Date.now()/400)*0.1;
        ctx.beginPath();ctx.arc(item.x,item.y,item.r,0,Math.PI*2);
        ctx.fillStyle=`rgba(255,255,255,${alpha})`;ctx.fill();
      }
    });
    // 绘制玩家
    const ab=save.team&&save.team.length>0?save.team[0]:save.activeBall;
    const def=BALL_DEFS[ab]||BALL_DEFS.thorn;
    const lv=getBallLevel(ab);
    ctx.beginPath();ctx.arc(worldPlayer.x,worldPlayer.y,worldPlayer.size,0,Math.PI*2);
    ctx.fillStyle=def.color;ctx.shadowColor=def.color;ctx.shadowBlur=12;ctx.fill();ctx.shadowBlur=0;
    ctx.fillStyle='#fff';ctx.font='bold 10px Arial';ctx.textAlign='center';
    ctx.fillText(def.name+(lv>0?` Lv${lv}`:''),worldPlayer.x,worldPlayer.y-worldPlayer.size-4);
    // 碰撞检测
    worldItems.forEach(item=>{
      if(item.collected)return;
      if(dist(worldPlayer,item)<worldPlayer.size+item.r){
        item.collected=true;
        if(item.type==='coin'){
          save.coins+=item.val;writeSave(save);updateHUD();showNotif('+'+item.val+' 金幣！');
        }else if(item.type==='chest') openChest();
        else if(item.type==='statue') openBlessing();
        else if(item.type==='buff')   pickupBuff(item.buffType);
        else if(item.type==='hidden') foundHidden();
      }
    });
    if(worldItems.every(i=>i.collected))spawnWorldItems();
    worldAnimId=requestAnimationFrame(tick);
  }
  tick();
}

function makeBattleBall(id,x,y,W,H){
  const def=ALL_DEFS[id]||BALL_DEFS.thorn;
  const angle=Math.random()*Math.PI*2;
  const hp=def.maxHp||100;
  return{id,hp,maxHp:hp,x,y,r:def.isBoss?26:18,
    vx:Math.cos(angle)*(def.speed||3),vy:Math.sin(angle)*(def.speed||3),
    skillTimer:0,skill2Timer:0,wolfCdTimer:0,
    burnTimer:0,gravTimer:0,auraTimer:0,
    stunTimer:0,petrifyTimer:0,
    dashCrit:false,spinTimer:0,
    thorns:[],_fireZones:[],W,H,isBoss:!!def.isBoss};
}

function updateBattleBall(ball,enemies,W,H){
  if(ball.hp<=0)return;
  const def=ALL_DEFS[ball.id]||BALL_DEFS.thorn;
  const alive=enemies.filter(e=>e&&e.hp>0);
  if(ball.stunTimer>0){ball.stunTimer-=16;if(ball.stunTimer<0)ball.stunTimer=0;}
  if(ball.petrifyTimer>0){ball.petrifyTimer-=16;return;}
  if(ball.stunTimer<=0){
    ball.x+=ball.vx;ball.y+=ball.vy;
    if(ball.x-ball.r<0){ball.x=ball.r;ball.vx*=-1;onBounce(ball);}
    if(ball.x+ball.r>W){ball.x=W-ball.r;ball.vx*=-1;onBounce(ball);}
    if(ball.y-ball.r<0){ball.y=ball.r;ball.vy*=-1;onBounce(ball);}
    if(ball.y+ball.r>H){ball.y=H-ball.r;ball.vy*=-1;onBounce(ball);}
  }
  if(ball.skillTimer>0)ball.skillTimer-=16;
  if(ball.wolfCdTimer>0)ball.wolfCdTimer-=16;
  // 接触伤害
  ball._contactTimer=(ball._contactTimer||0)+16;
  if(ball._contactTimer>=600){
    ball._contactTimer=0;
    alive.forEach(en=>{
      if(dist(ball,en)<ball.r+en.r+2){
        const mul=getElementMul(ball.id,en.id);
        dealDamage(en,(def.dmg||1)*mul,ball);
        dealDamage(ball,(ALL_DEFS[en.id]?.dmg||1),en);
      }
    });
  }
  switch(def.id){
    case 'thorn':       handleThorn(ball,alive);break;
    case 'sunflash':    handleAura(ball,alive,def.auraDmg,def.auraRadius,def.auraInterval);break;
    case 'blackhole':   handleGravity(ball,alive,def);break;
    case 'giant':       handleGiant(ball,alive,def);break;
    case 'cat':         handleCat(ball,alive,def);break;
    case 'wolf':        handleWolf(ball,alive,def);break;
    case 'fireball':    handleFireball(ball,alive,def);break;
    case 'dragon':      handleDragon(ball,alive,def);break;
    case 'vampire':     handleVampire(ball,alive,def);break;
    case 'onepunch':    handleOnePunch(ball,alive,def);break;
    case 'swordmaster': handleSword(ball,alive,def);break;
    case 'magician':    handleMagician(ball,alive,def);break;
    case 'thunder':     handleThunder(ball,alive,def);break;
    case 'rock':        handleRock(ball,alive,def);break;
    case 'windassassin':handleWind(ball,alive,def);break;
    case 'boss5':       handleBoss5(ball,alive,def);break;
    case 'boss10':      handleBoss10(ball,alive,def);break;
    case 'boss15':      handleBoss15(ball,alive,def);break;
  }
}

function onBounce(ball){
  if(ball.id==='thorn')ball.thorns.push({x:ball.x,y:ball.y,life:8000});
  if(ball.id==='fireball'){ball._fireZones=ball._fireZones||[];if(ball._fireZones.length<5)ball._fireZones.push({x:ball.x,y:ball.y,r:28,life:4000,_dmgTimer:0});}
}

// AI技能处理函数
function handleThorn(ball,en){ball.thorns.forEach(t=>{en.forEach(e=>{if(dist(e,t)<e.r+20)dealDamage(e,2,ball);});t.life-=16;});ball.thorns=ball.thorns.filter(t=>t.life>0);}
function handleAura(ball,en,dmg,r,iv){ball.auraTimer=(ball.auraTimer||0)+16;if(ball.auraTimer>=iv){ball.auraTimer=0;en.forEach(e=>{if(dist(ball,e)<r)dealDamage(e,dmg,ball);});}}
function handleGravity(ball,en,def){en.forEach(e=>{if(dist(ball,e)<def.gravRadius){const dx=ball.x-e.x,dy=ball.y-e.y,d=Math.hypot(dx,dy)||1;e.vx+=dx/d*0.5;e.vy+=dy/d*0.5;ball.gravTimer=(ball.gravTimer||0)+16;if(ball.gravTimer>=1000){ball.gravTimer-=1000;dealDamage(e,def.gravDmg,ball);}}});}
function handleGiant(ball,en,def){if(ball.spinTimer>0){ball.spinTimer-=16;ball._spinDmgTimer=(ball._spinDmgTimer||0)+16;if(ball._spinDmgTimer>=300){ball._spinDmgTimer=0;en.forEach(e=>{if(dist(ball,e)<ball.r+e.r+22)dealDamage(e,def.skillDmg*0.1,ball);});}return;}ball._meleeDmgTimer=(ball._meleeDmgTimer||0)+16;if(ball._meleeDmgTimer>=500){ball._meleeDmgTimer=0;en.forEach(e=>{if(dist(ball,e)<ball.r+e.r+5)dealDamage(e,def.dmg,ball);});}if(ball.skillTimer<=0){ball.skillTimer=def.skillCd*1000;ball.spinTimer=3000;}}
function handleCat(ball,en,def){const enemy=en[0];ball._blinkTimer=(ball._blinkTimer||2500)-16;if(ball._blinkTimer<=0&&enemy){ball._blinkTimer=2500;const a=Math.random()*Math.PI*2,off=rand(30,60);ball.x=Math.max(ball.r,Math.min(ball.W-ball.r,enemy.x+Math.cos(a)*off));ball.y=Math.max(ball.r,Math.min(ball.H-ball.r,enemy.y+Math.sin(a)*off));const dx=enemy.x-ball.x,dy=enemy.y-ball.y,d=Math.hypot(dx,dy)||1;ball.vx=(dx/d)*def.speed;ball.vy=(dy/d)*def.speed;}ball._clawTimer=(ball._clawTimer||0)+16;if(ball._clawTimer>=1500){ball._clawTimer=0;en.forEach(e=>{if(dist(ball,e)<ball.r+e.r+30)dealDamage(e,def.dmg,ball);});}if(ball.skillTimer<=0){ball.skillTimer=def.skillCd*1000;en.forEach(e=>dealDamage(e,def.skillDmg,ball));}}
function handleWolf(ball,en,def){en.forEach(e=>{if(ball.wolfCdTimer<=0&&dist(ball,e)<ball.r+e.r+4){dealDamage(e,def.dmg,ball);ball.wolfCdTimer=def.skillCd*1000;}});}
function handleFireball(ball,en,def){en.forEach(e=>{const t=dist(ball,e)<ball.r+e.r+2;if(t){ball._burnLinger=2000;ball.burnTimer=(ball.burnTimer||0)+16;if(ball.burnTimer>=def.burnInterval){ball.burnTimer=0;dealDamage(e,def.burnDmg,ball);}}else{if(ball._burnLinger>0){ball._burnLinger-=16;ball.burnTimer=(ball.burnTimer||0)+16;if(ball.burnTimer>=def.burnInterval){ball.burnTimer=0;dealDamage(e,def.burnDmg*0.6,ball);}}else ball.burnTimer=0;}});if(ball._explodeTimer===undefined)ball._explodeTimer=8000;ball._explodeTimer-=16;if(ball._explodeTimer<=0){ball._explodeTimer=8000;ball._explodeAnim=400;en.forEach(e=>{if(dist(ball,e)<120)dealDamage(e,20,ball);});}if(ball._explodeAnim>0)ball._explodeAnim-=16;ball._fireZones=ball._fireZones||[];ball._fireZones.forEach(z=>{z.life-=16;z._dmgTimer=(z._dmgTimer||0)+16;if(z._dmgTimer>=300){z._dmgTimer=0;en.forEach(e=>{if(dist(e,z)<z.r+e.r)dealDamage(e,3,ball);});}});ball._fireZones=ball._fireZones.filter(z=>z.life>0);}
function handleDragon(ball,en,def){ball._spikeTimer=(ball._spikeTimer||0)+16;if(ball._spikeTimer>=1000){ball._spikeTimer=0;en.forEach(e=>dealDamage(e,def.dmg,ball));}if(ball.skillTimer<=0){ball.skillTimer=def.skillCd*1000;en.forEach(e=>dealDamage(e,def.skillDmg,ball));}}

const particles=[],dmgNumbers=[];
function spawnParticles(x,y,color,count,opt={}){
  for(let i=0;i<count;i++){
    const a=Math.random()*Math.PI*2,spd=rand(opt.minSpeed||2,opt.maxSpeed||7);
    particles.push({x,y,vx:Math.cos(a)*spd,vy:Math.sin(a)*spd,
      life:opt.life||rand(400,700),maxLife:opt.life||550,
      size:rand(opt.minSize||3,opt.maxSize||9),
      color:opt.color||color,shape:opt.shape||'circle',gravity:opt.gravity||0});
  }
}
function spawnDmgNumber(x,y,dmg,isElem=false){
  dmgNumbers.push({x,y:y-10,vy:-2.2,life:700,maxLife:700,
    text:'-'+Math.ceil(dmg),color:isElem?'#ffaa00':'#ff2222'});
}
function updateParticles(){
  for(let i=particles.length-1;i>=0;i--){
    const p=particles[i];p.x+=p.vx;p.y+=p.vy;p.vy+=p.gravity;p.vx*=0.94;p.vy*=0.94;p.life-=16;
    if(p.life<=0)particles.splice(i,1);
  }
  for(let i=dmgNumbers.length-1;i>=0;i--){
    const d=dmgNumbers[i];d.y+=d.vy;d.vy*=0.97;d.life-=16;
    if(d.life<=0)dmgNumbers.splice(i,1);
  }
}
function drawParticles(ctx){
  particles.forEach(p=>{
    const alpha=Math.max(0,p.life/p.maxLife);
    ctx.save();ctx.globalAlpha=alpha;ctx.fillStyle=p.color;ctx.shadowColor=p.color;ctx.shadowBlur=8;
    if(p.shape==='star'){
      ctx.beginPath();
      for(let s=0;s<10;s++){const a=(s*Math.PI)/5-Math.PI/2,r=s%2===0?p.size:p.size*0.4;
        s===0?ctx.moveTo(p.x+Math.cos(a)*r,p.y+Math.sin(a)*r):ctx.lineTo(p.x+Math.cos(a)*r,p.y+Math.sin(a)*r);}
      ctx.closePath();ctx.fill();
    }else if(p.shape==='spark'){
      ctx.strokeStyle=p.color;ctx.lineWidth=Math.max(1,p.size*0.6);
      ctx.beginPath();ctx.moveTo(p.x,p.y);ctx.lineTo(p.x-p.vx*4,p.y-p.vy*4);ctx.stroke();
    }else{ctx.beginPath();ctx.arc(p.x,p.y,p.size,0,Math.PI*2);ctx.fill();}
    ctx.restore();
  });
  dmgNumbers.forEach(d=>{
    const alpha=Math.max(0,d.life/d.maxLife),scale=1+(1-d.life/d.maxLife)*0.4;
    ctx.save();ctx.globalAlpha=alpha;
    ctx.font=`bold ${Math.round(15*scale)}px Arial`;ctx.textAlign='center';
    ctx.strokeStyle='#000';ctx.lineWidth=3;ctx.strokeText(d.text,d.x,d.y);
    ctx.fillStyle=d.color||'#ff2222';ctx.fillText(d.text,d.x,d.y);
    ctx.restore();
  });
}

// DEAL DAMAGE — 含属性克制 + 等级加成 + 临时buff
// ============================================================
function dealDamage(target,amount,source){
  if(target.hp<=0||target.petrifyTimer>0)return;
  let dmg=amount;
  // 资质倍率
  if(source&&source._dmgMul&&source._dmgMul!==1)dmg*=source._dmgMul;
  if(source&&source._atkBonus)dmg+=source._atkBonus;
  // 等级加成
  if(source&&source.isPlayer){
    const lvBonus=getBallLevelBonus(source.id);
    dmg*=(1+lvBonus.atkPct);
  }
  // 属性克制
  let isElem=false;
  if(source&&target){
    const mul=getElementMul(source.id,target.id);
    if(mul>1){dmg*=mul;isElem=true;}
  }
  // 临时buff：伤害加成
  if(source&&source.isPlayer&&save._tempBuffs?.dmgUp&&Date.now()<save._tempBuffs.dmgUp){
    dmg*=1.5;
  }
  // 护盾：玩家无敌
  if(target.isPlayer&&save._tempBuffs?.shield&&Date.now()<save._tempBuffs.shield){
    return;
  }
  // 磐石减伤
  if(target.id==='rock')dmg=Math.max(0,dmg-(BALL_DEFS.rock.armor||2));
  target.hp=Math.max(0,target.hp-dmg);
  if(source&&dmg>0){
    spawnDmgNumber(target.x+rand(-16,16),target.y-target.r-rand(3,10),dmg,isElem);
    const x=target.x,y=target.y;
    const sc=source.id;
    if(sc==='swordmaster')spawnParticles(x,y,'#00ddff',12,{shape:'spark',minSpeed:5,maxSpeed:12,life:300,minSize:2,maxSize:4});
    else if(sc==='vampire')spawnParticles(x,y,'#ff0044',10,{minSpeed:2,maxSpeed:6,life:450,minSize:3,maxSize:7});
    else if(sc==='magician')spawnParticles(x,y,'#ff44ff',14,{shape:'star',minSpeed:3,maxSpeed:8,life:550,minSize:4,maxSize:9});
    else if(sc==='fireball')spawnParticles(x,y,'#ff4400',12,{minSpeed:3,maxSpeed:8,life:450,minSize:3,maxSize:8,gravity:0.15});
    else if(sc==='thunder')spawnParticles(x,y,'#ffff00',14,{shape:'spark',minSpeed:6,maxSpeed:14,life:220,minSize:2,maxSize:4});
    else if(sc==='giant'||sc==='boss5')spawnParticles(x,y,'#ff4400',16,{minSpeed:3,maxSpeed:9,life:500,minSize:5,maxSize:12,gravity:0.2});
    else if(sc==='onepunch')spawnParticles(x,y,'#fff',28,{minSpeed:6,maxSpeed:18,life:650,minSize:5,maxSize:13});
    else if(sc==='boss10')spawnParticles(x,y,'#880088',16,{shape:'star',minSpeed:3,maxSpeed:9,life:600,minSize:4,maxSize:10});
    else if(sc==='boss15')spawnParticles(x,y,'#ffffff',20,{shape:'star',minSpeed:5,maxSpeed:14,life:700,minSize:5,maxSize:12});
    else spawnParticles(x,y,ALL_DEFS[sc]?.color||'#fff',8,{minSpeed:2,maxSpeed:6,life:320,minSize:3,maxSize:7});
  }
}

    const pb=makeBattleBall(id,W*0.2,H*0.5,W,H);
    pb.isPlayer=true;pb.vx=0;pb.vy=0;
    pb.hp=Math.floor(baseHp);pb.maxHp=Math.floor(baseHp);
    pb._dmgMul=boost.dmgMul||1;
    pb._atkBonus=boost.atkBonus||0;
    pb._atkCd=0;pb._atkCdMax=800;
    // CD减半buff
    const cdMulBuff=(save._tempBuffs?.cdHalf&&Date.now()<save._tempBuffs.cdHalf)?0.5:1;
    pb._skillCd=0;pb._skillCdMax=Math.max(500,(def.skillCd||10)*1000*cdMulBuff);
    pb._alive=true;
    return pb;
  }).filter(Boolean);
  arenaActiveIdx=0;
  switchCd=0;

  // 创建敌人
  const enemyBalls=enemyIds.map((id,i)=>{
    const def=ALL_DEFS[id];if(!def)return null;
    const ex=enemyIds.length>1?W*0.7+i*100:W*0.75;
    const ey=enemyIds.length>1?H*0.35+i*130:H*0.5;
    const eb=makeBattleBall(id,ex,ey,W,H);
    eb.isEnemy=true;
    // 难度加成
    eb.hp=Math.floor(eb.hp*(isBossStage?1:1));
    eb.maxHp=eb.hp;
    eb.vx*=modeMult.speed;eb.vy*=modeMult.speed;
    eb._modeDmgMul=modeMult.dmg;
    eb._skillCdMul=modeMult.cdMul;
    return eb;
  }).filter(Boolean);

  arenaState={enemyBalls,stage,done:false,isBossStage,mode};

  // 标题
  let stageLabel=`第${stage}關`;
  if(isBossStage)stageLabel+=` 🔥BOSS`;
  if(mode==='hard')stageLabel+=' [困難]';
  if(mode==='hell')stageLabel+=' [地獄]';
  stageLabel+=` — ${enemyIds.map(id=>ALL_DEFS[id]?.name||id).join(' + ')}`;
  document.getElementById('arenaInfo').textContent=stageLabel;
  document.getElementById('arenaResult').style.display='none';

  // UI
  const mobile=isMobile();
  document.getElementById('mobileControls').style.display=mobile?'block':'none';
  document.getElementById('cdPanel').classList.add('show');
  document.getElementById('keyHint').style.display=mobile?'none':'block';
  if(mobile){['cdKeyAtk'].forEach(id=>{const el=document.getElementById(id);if(el)el.textContent='';})}

  // BOSS提示
  if(isBossStage){
    const alert=document.createElement('div');
    alert.className='boss-alert';
    alert.textContent='⚠️ BOSS戰！';
    document.getElementById('arenaScreen').appendChild(alert);
    setTimeout(()=>alert.remove(),2000);
  }

  buildTeamBar();
  setupMobileControls();
  arenaLoop(canvas);
}

// ── 队伍切换栏 ────────────────────────────────────────────────
function buildTeamBar(){
  const bar=document.getElementById('teamBar');
  bar.innerHTML='';
  if(arenaTeamBalls.length<=1){bar.classList.remove('show');return;}
  bar.classList.add('show');
  arenaTeamBalls.forEach((pb,i)=>{
    const def=BALL_DEFS[pb.id];if(!def)return;
    const slot=document.createElement('div');
    slot.className='team-slot'+(i===arenaActiveIdx?' active-slot':'')+(pb.hp<=0?' dead-slot':'');
    slot.style.background=def.color+'44';
    slot.style.borderColor=i===arenaActiveIdx?'#00ff88':'#444';
    slot.innerHTML=
      `<span style="color:${def.color};font-size:10px">${def.name}</span>`+
      `<div class="slot-hp">${Math.ceil(pb.hp)}/${pb.maxHp}</div>`;
    slot.onclick=()=>{if(pb.hp>0&&i!==arenaActiveIdx)doSwitch(i);};
    bar.appendChild(slot);
  });
}

function doSwitch(idx){
  if(switchCd>0){showNotif(`換球CD：${(switchCd/1000).toFixed(1)}s`,'#ff8800');return;}
  if(!arenaTeamBalls[idx]||arenaTeamBalls[idx].hp<=0)return;
  arenaActiveIdx=idx;
  switchCd=SWITCH_CD_MAX;
  buildTeamBar();
  showNotif(`🔄 切換為【${ballName(arenaTeamBalls[idx].id)}】！`,'#ff8800');
}

// ── 手机摇杆 ──────────────────────────────────────────────────
function setupMobileControls(){
  const zone=document.getElementById('joystickZone');
  const knob=document.getElementById('joystickKnob');
  zone.ontouchstart=e=>{
    e.preventDefault();
    const t=e.changedTouches[0],rect=zone.getBoundingClientRect();
    joy.active=true;joy.touchId=t.identifier;
    joy.dx=t.clientX-rect.left-64;joy.dy=t.clientY-rect.top-64;
    clampJoy();updateKnob();
  };
  zone.ontouchmove=e=>{
    e.preventDefault();
    for(const t of e.changedTouches){
      if(t.identifier!==joy.touchId)continue;
      const rect=zone.getBoundingClientRect();
      joy.dx=t.clientX-rect.left-64;joy.dy=t.clientY-rect.top-64;
      clampJoy();updateKnob();
    }
  };
  zone.ontouchend=zone.ontouchcancel=()=>{
    joy.active=false;joy.dx=0;joy.dy=0;
    knob.style.transform='translate(0,0)';
  };
  function clampJoy(){const len=Math.hypot(joy.dx,joy.dy),max=40;if(len>max){joy.dx=joy.dx/len*max;joy.dy=joy.dy/len*max;}}
  function updateKnob(){knob.style.transform=`translate(${joy.dx}px,${joy.dy}px)`;}

  const mBtnAtk=document.getElementById('mBtnAtk');
  const mBtnSkill=document.getElementById('mBtnSkill');
  const mBtnSwitch=document.getElementById('mBtnSwitch');
  mBtnAtk.ontouchstart=e=>{e.preventDefault();atkPressed=true;};
  mBtnAtk.ontouchend=mBtnAtk.ontouchcancel=e=>{e.preventDefault();atkPressed=false;};
  mBtnSkill.ontouchstart=e=>{e.preventDefault();skillPressed=true;};
  mBtnSkill.ontouchend=mBtnSkill.ontouchcancel=e=>{e.preventDefault();skillPressed=false;};
  mBtnSwitch.ontouchstart=e=>{
    e.preventDefault();
    // 找下一个存活球
    const next=findNextAlive();
    if(next!==-1)doSwitch(next);
  };
}

function findNextAlive(){
  for(let i=1;i<=arenaTeamBalls.length;i++){
    const idx=(arenaActiveIdx+i)%arenaTeamBalls.length;
    if(arenaTeamBalls[idx]&&arenaTeamBalls[idx].hp>0&&idx!==arenaActiveIdx)return idx;
  }
  return -1;
}

// ── 玩家手动更新 ──────────────────────────────────────────────
function updatePlayerManual(pb,enemies,W,H,dt){
  if(!pb||pb.hp<=0||arenaState.done)return;
  const SPEED=5.5;
  let mx=0,my=0;
  if(aKeys.up)   my-=1;
  if(aKeys.down) my+=1;
  if(aKeys.left) mx-=1;
  if(aKeys.right)mx+=1;
  if(joy.active&&Math.hypot(joy.dx,joy.dy)>6){mx=joy.dx/40;my=joy.dy/40;}
  const mlen=Math.hypot(mx,my);
  if(mlen>0){mx/=mlen;my/=mlen;}
  pb.vx=mx*SPEED;pb.vy=my*SPEED;

  if(pb.stunTimer<=0&&pb.petrifyTimer<=0){
    pb.x+=pb.vx;pb.y+=pb.vy;
    pb.x=Math.max(pb.r,Math.min(W-pb.r,pb.x));
    pb.y=Math.max(pb.r,Math.min(H-pb.r,pb.y));
  }
  if(pb.stunTimer>0){pb.stunTimer=Math.max(0,pb.stunTimer-dt);}
  if(pb.petrifyTimer>0){pb.petrifyTimer=Math.max(0,pb.petrifyTimer-dt);}

  // debuff效果
  if(currentDebuff&&currentDebuff.onUpdate)currentDebuff.onUpdate(pb);

  if(pb._atkCd>0)  pb._atkCd  =Math.max(0,pb._atkCd-dt);
  if(pb._skillCd>0)pb._skillCd=Math.max(0,pb._skillCd-dt);
  if(switchCd>0)   switchCd   =Math.max(0,switchCd-dt);

  const alive=enemies.filter(e=>e&&e.hp>0);

  // 攻击
  if(atkPressed&&pb._atkCd<=0){
    pb._atkCd=pb._atkCdMax;atkPressed=false;
    const def=BALL_DEFS[pb.id];
    const target=alive.sort((a,b)=>dist(pb,a)-dist(pb,b))[0];
    if(target){
      const mul=getElementMul(pb.id,target.id);
      dealDamage(target,(def.dmg||5)*mul,pb);
      const dx=target.x-pb.x,dy=target.y-pb.y,d=Math.hypot(dx,dy)||1;
      target.vx+=dx/d*4;target.vy+=dy/d*4;
    }
  }

  // 技能
  if(skillPressed&&pb._skillCd<=0){
    pb._skillCd=pb._skillCdMax;skillPressed=false;
    triggerPlayerSkill(pb,alive);
  }

  // Tab切换
  if(switchPressed){
    switchPressed=false;
    const next=findNextAlive();
    if(next!==-1)doSwitch(next);
  }

  // 被动
  handlePlayerPassive(pb,alive);

  // 接触伤害（敌人碰玩家）
  pb._contactTimer=(pb._contactTimer||0)+dt;
  if(pb._contactTimer>=800){
    pb._contactTimer=0;
    alive.forEach(e=>{
      if(dist(pb,e)<pb.r+e.r+2){
        const modeMult=MODE_MULT[save.gameMode||'normal']||MODE_MULT.normal;
        const dmg=(ALL_DEFS[e.id]?.dmg||1)*modeMult.dmg*(e._modeDmgMul||1);
        dealDamage(pb,dmg,e);
      }
    });
  }
}

// ── 玩家技能 ──────────────────────────────────────────────────
function triggerPlayerSkill(pb,alive){
  const def=BALL_DEFS[pb.id];if(!def)return;
  const target=alive.sort((a,b)=>dist(pb,a)-dist(pb,b))[0];
  const lvBonus=getBallLevelBonus(pb.id);
  const skillMul=1+lvBonus.skillPct;
  switch(def.id){
    case 'thorn':
      for(let i=0;i<8;i++){const a=(i/8)*Math.PI*2;pb.thorns.push({x:pb.x+Math.cos(a)*50,y:pb.y+Math.sin(a)*50,life:6000});}
      showNotif('🌿 毒刺環！','#44ff44');break;
    case 'sunflash':
      alive.forEach(e=>dealDamage(e,15*skillMul,pb));
      showNotif('☀️ 太陽爆發！','#ffdd00');break;
    case 'blackhole':
      alive.forEach(e=>{const dx=pb.x-e.x,dy=pb.y-e.y,d=Math.hypot(dx,dy)||1;e.vx+=dx/d*14;e.vy+=dy/d*14;dealDamage(e,8*skillMul,pb);});
      showNotif('🌀 黑洞引力！','#6600cc');break;
    case 'giant':
      pb.spinTimer=3000;showNotif('🔨 旋轉巨錘！','#cc8800');break;
    case 'cat':
      if(target){const a=Math.random()*Math.PI*2;pb.x=Math.max(pb.r,Math.min(pb.W-pb.r,target.x+Math.cos(a)*40));pb.y=Math.max(pb.r,Math.min(pb.H-pb.r,target.y+Math.sin(a)*40));dealDamage(target,def.skillDmg*skillMul,pb);}
      showNotif('🐱 瞬移追蹤！','#ff88cc');break;
    case 'wolf':
      alive.forEach(e=>dealDamage(e,16*skillMul,pb));
      showNotif('🐺 狂暴嚎叫！','#aaa');break;
    case 'fireball':
      alive.forEach(e=>{if(dist(pb,e)<160)dealDamage(e,20*skillMul,pb);});pb._explodeAnim=400;
      showNotif('💥 爆炸！','#ff4400');break;
    case 'dragon':
      alive.forEach(e=>dealDamage(e,(def.skillDmg||5)*skillMul,pb));
      showNotif('🐉 石球衝擊！','#0088ff');break;
    case 'vampire':
      if(target){pb._vampBound=true;pb._vampDmgTimer=0;pb._vampBindRemain=(def.bindDur||3)*1000;target.stunTimer=(def.bindDur||3)*1000;}
      showNotif('🧛 束縛！','#880000');break;
    case 'onepunch':
      if(target){if(Math.random()>def.missChance)dealDamage(target,def.skillDmg*skillMul,pb);else showNotif('😅 打空了！','#888');}
      else showNotif('👊 一拳！','#fff');break;
    case 'swordmaster':
      if(target){pb._swordBound=true;pb._swordDmgTimer=0;pb._swordBindRemain=(def.bindDur||4)*1000;target.stunTimer=(def.bindDur||4)*1000;}
      showNotif('⚔️ 旋轉劍束縛！','#00ddff');break;
    case 'magician':
      alive.forEach(e=>{dealDamage(e,(def.allDmg||10)*skillMul,pb);e.stunTimer=2000;});
      showNotif('🎩 全場衝擊！','#ff44ff');break;
    case 'thunder':
      pb._fieldTimer=(def.fieldDur||2)*1000;alive.forEach(e=>dealDamage(e,(def.dmg||7)*skillMul,pb));
      showNotif('⚡ 雷電領域！','#ffff00');break;
    case 'rock':
      pb.petrifyTimer=1500;
      setTimeout(()=>{alive.forEach(e=>{if(e.hp>0)dealDamage(e,(def.skillDmg||10)*skillMul,pb);});},1500);
      showNotif('🪨 無敵衝撞！','#888866');break;
    case 'windassassin':
      if(target){const dx=target.x-pb.x,dy=target.y-pb.y,d=Math.hypot(dx,dy)||1;pb.x+=dx/d*90;pb.y+=dy/d*90;pb.dashCrit=true;dealDamage(target,(def.critDmg||10)*skillMul,pb);}
      showNotif('💨 閃避暴擊！','#88ffcc');break;
    default:
      alive.forEach(e=>dealDamage(e,10*skillMul,pb));showNotif('💥 技能！','#fff');
  }
}

// ── 玩家被动 ──────────────────────────────────────────────────
function handlePlayerPassive(pb,alive){
  const def=BALL_DEFS[pb.id];if(!def)return;
  if(def.id==='thorn')    handleThorn(pb,alive);
  if(def.id==='sunflash') handleAura(pb,alive,def.auraDmg,def.auraRadius,def.auraInterval);
  if(def.id==='blackhole')handleGravity(pb,alive,def);
  if(def.id==='giant'&&pb.spinTimer>0){
    pb.spinTimer-=16;pb._spinDmgTimer=(pb._spinDmgTimer||0)+16;
    if(pb._spinDmgTimer>=300){pb._spinDmgTimer=0;alive.forEach(e=>{if(dist(pb,e)<pb.r+e.r+20)dealDamage(e,BALL_DEFS.giant.skillDmg*0.1,pb);});}
  }
  if(def.id==='vampire'&&pb._vampBound)handleVampire(pb,alive,def);
  if(def.id==='swordmaster'&&pb._swordBound)handleSword(pb,alive,def);
  if(def.id==='thunder'&&pb._fieldTimer>0){
    pb._fieldTimer-=16;pb._fieldDmgT=(pb._fieldDmgT||0)+16;
    if(pb._fieldDmgT>=1000){pb._fieldDmgT=0;alive.forEach(e=>{if(dist(pb,e)<def.fieldRadius)dealDamage(e,def.fieldDmg,pb);});}
  }
  if(def.id==='fireball')handleFireball(pb,alive,def);
}

// ── CD圆环 ────────────────────────────────────────────────────
function drawCdRing(canvasId,cdNow,cdMax){
  const c=document.getElementById(canvasId);if(!c)return;
  const ctx=c.getContext('2d');
  ctx.clearRect(0,0,58,58);
  if(cdNow>0){
    const pct=cdNow/cdMax;
    ctx.beginPath();ctx.moveTo(29,29);
    ctx.arc(29,29,27,-Math.PI/2,-Math.PI/2+Math.PI*2*pct);
    ctx.closePath();ctx.fillStyle='rgba(0,0,0,0.65)';ctx.fill();
    ctx.fillStyle='#fff';ctx.font='bold 12px Arial';ctx.textAlign='center';ctx.textBaseline='middle';
    ctx.fillText(Math.ceil(cdNow/1000)+'s',29,29);
  }
}

// ── 主循环 ────────────────────────────────────────────────────
function arenaLoop(canvas){
  const ctx=canvas.getContext('2d');
  const W=canvas.width,H=canvas.height;
  const {enemyBalls}=arenaState;
  let lastTime=performance.now();

  function tick(now){
    const dt=Math.min(now-lastTime,50);lastTime=now;
    ctx.clearRect(0,0,W,H);
    ctx.fillStyle='#0a0a0a';ctx.fillRect(0,0,W,H);

    // 边框颜色根据模式
    const borderColors={normal:'#ff8800',hard:'#ff3300',hell:'#aa00ff'};
    ctx.strokeStyle=borderColors[save.gameMode||'normal']||'#ff8800';
    ctx.lineWidth=4;ctx.strokeRect(4,4,W-8,H-8);

    // 网格
    ctx.strokeStyle='#ffffff08';ctx.lineWidth=1;
    for(let gx=0;gx<W;gx+=60){ctx.beginPath();ctx.moveTo(gx,0);ctx.lineTo(gx,H);ctx.stroke();}
    for(let gy=0;gy<H;gy+=60){ctx.beginPath();ctx.moveTo(0,gy);ctx.lineTo(W,gy);ctx.stroke();}

    // debuff全场tick
    debuffTimer+=dt;
    if(currentDebuff?.id==='poison'&&debuffTimer>=3000){
      debuffTimer=0;
      const allBalls=[...arenaTeamBalls,...enemyBalls];
      if(currentDebuff.onTick)currentDebuff.onTick(allBalls);
    }

    const pb=arenaTeamBalls[arenaActiveIdx];

    if(!arenaState.done){
      // 更新当前操控球
      if(pb&&pb.hp>0)updatePlayerManual(pb,enemyBalls,W,H,dt);
      // 更新其他队伍球（静止等待）
      arenaTeamBalls.forEach((b,i)=>{
        if(i!==arenaActiveIdx&&b&&b.hp>0){
          // 非操控球缩到角落待机
          const targetX=40+i*70,targetY=H-40;
          b.x+=(targetX-b.x)*0.05;b.y+=(targetY-b.y)*0.05;
          b.vx=0;b.vy=0;
        }
      });
      // 更新敌人
      enemyBalls.forEach(eb=>{
        if(eb&&eb.hp>0){
          // debuff对敌人也生效
          if(currentDebuff?.onUpdate)currentDebuff.onUpdate(eb);
          updateBattleBall(eb,arenaTeamBalls.filter(b=>b&&b.hp>0),W,H);
        }
      });
    }

    // 绘制所有球
    arenaTeamBalls.forEach((b,i)=>{
      if(!b||b.hp<=0)return;
      drawBattleBall(ctx,b);
      // 当前操控球高亮
      if(i===arenaActiveIdx){
        ctx.beginPath();ctx.arc(b.x,b.y,b.r+5,0,Math.PI*2);
        ctx.strokeStyle='#00ff8866';ctx.lineWidth=3;ctx.stroke();
      }
    });
    enemyBalls.forEach(eb=>{if(eb)drawBattleBall(ctx,eb);});
    updateParticles();drawParticles(ctx);

    // HP条
    if(pb&&pb.hp>=0){
      const pPct=Math.max(0,pb.hp/pb.maxHp)*100;
      document.getElementById('playerHpFill').style.width=pPct+'%';
      // 显示当前球+队伍信息
      const teamInfo=arenaTeamBalls.map((b,i)=>b?`${i===arenaActiveIdx?'▶':' '}${Math.ceil(Math.max(0,b.hp))}`:'✕').join(' ');
      document.getElementById('playerHpLabel').textContent=`我方：${teamInfo}`;
    }
    const tEHp=enemyBalls.reduce((s,e)=>s+(e?Math.max(0,e.hp):0),0);
    const tEMax=enemyBalls.reduce((s,e)=>s+(e?e.maxHp:0),0);
    document.getElementById('enemyHpFill').style.width=(tEMax>0?tEHp/tEMax*100:0)+'%';
    document.getElementById('enemyHpLabel').textContent=
      enemyBalls.length>1
        ?`敵方：${enemyBalls.map(e=>e?Math.ceil(Math.max(0,e.hp)):0).join('+')}`
        :`敵方：${enemyBalls[0]?Math.ceil(Math.max(0,enemyBalls[0].hp)):0}`;

    // CD圆环
    if(pb){
      drawCdRing('cdCanvasAtk',  pb._atkCd,   pb._atkCdMax);
      drawCdRing('cdCanvasSkill',pb._skillCd, pb._skillCdMax);
      drawCdRing('cdCanvasSwitch',switchCd,   SWITCH_CD_MAX);
      updateMobileCdText(pb);
    }

    // 更新队伍栏
    buildTeamBar();

    // 胜负判断
    if(!arenaState.done){
      const allPlayerDead=arenaTeamBalls.every(b=>!b||b.hp<=0);
      const allEnemyDead=enemyBalls.every(e=>!e||e.hp<=0);
      if(allPlayerDead||allEnemyDead){
        arenaState.done=true;
        const won=allEnemyDead;
        const res=document.getElementById('arenaResult');
        res.style.display='block';
        if(won){
          res.innerHTML='🏆 勝利！';res.style.color='#ffcc00';
          const cs=save.arenaStage||1;
          const modeMult=MODE_MULT[save.gameMode||'normal']||MODE_MULT.normal;
          save.arenaStage=cs+1;
          save.coins+=modeMult.rewardCoins;
          save.qualityFruits=(save.qualityFruits||0)+modeMult.rewardFruits;
          save.totalWins=(save.totalWins||0)+1;
          // 世界事件额外奖励
          if(worldEventActive){
            save.qualityFruits+=2;save.coins+=30;
            showNotif('⚡ 入侵獎勵！額外+2果+30幣','#ffcc00');
          }
          // 第10关全局强化
          if(cs===10){
            save.ballBoost.dmgMul=+((save.ballBoost.dmgMul+0.3).toFixed(2));
            save.ballBoost.hpBonus=+((save.ballBoost.hpBonus+30).toFixed(0));
            showNotif('🎉 第10關！傷害+30% 生命+30','#ffcc00');
          }
          // 解锁困难/地狱
          if(save.totalWins>=15&&!save.unlockedHard){
            save.unlockedHard=true;showNotif('🔓 解鎖困難模式！','#ff4400');
          }
          if(save.totalWins>=30&&!save.unlockedHell){
            save.unlockedHell=true;showNotif('💀 解鎖地獄模式！','#aa00ff');
          }
          writeSave(save);updateHUD();
          if(cs===15)setTimeout(()=>triggerVictory(),1800);
          else       setTimeout(()=>_doEnterArena(),2800);
        }else{
          res.innerHTML='💀 全滅...<br><span style="font-size:16px;color:#aaa">隊伍全滅</span>';
          res.style.color='#ff4444';
        }
      }
    }
    arenaAnimId=requestAnimationFrame(tick);
  }
  arenaAnimId=requestAnimationFrame(tick);
}
